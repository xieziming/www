import java.io.BufferedReader;  
import java.io.IOException;
import java.io.InputStreamReader;  
import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.Date;
 
import javax.swing.JOptionPane;
 
public class  QTPMonitor {  
	public static final String processName="KMPlayer.exe";
	public static final Double minCpu= 2.0;
	public static final int minNet=4;
	public static final int lastMinutes=1;
	public String previousCpuTime;
	public boolean startedFlag;
	public QTPMonitor(){
		previousCpuTime="";
		startedFlag = false;
	}
 
	public void start(){
		if(!startedFlag){
			monitor();
		}else{
			JOptionPane.showMessageDialog(null, "QTP Monitor has alread been started!","Start QTP Monitor",JOptionPane.INFORMATION_MESSAGE);
		}
 
	}
	public void monitor(){
		while(true){
			if(Operations.findTaskByName(processName)){
				if(Operations.getTaskCPU(processName) < minCpu){
					if(previousCpuTime ==""){
						previousCpuTime = getTime();
					}else{
						String currentTime = getTime();
						long minutesNet = getTimeNet(previousCpuTime,currentTime)/(1000 * 60);
						System.out.println(previousCpuTime+" - "+currentTime+" :"+getTimeNet(previousCpuTime,currentTime));
						if(minutesNet >= lastMinutes){
							System.out.println("It seems that the process has dead! I am to kill it!");
							boolean killFlag = false;
							while(!killFlag){
								killFlag = Operations.killTask(processName);
								if(killFlag){
									System.out.println("task of "+processName+" has been killed!");
								}else{
									System.out.println("task of "+processName+" fail to be killed! retry");
								}
							}
							previousCpuTime = getTime();
						}
					}
				}else{
					previousCpuTime = getTime();
				}
			}else{
				previousCpuTime = getTime();
				System.out.println("process " + processName+" doesn't exist");
			}
 
			try {
				Thread.sleep(5000);
			} catch (InterruptedException e) {
				e.printStackTrace();
			}
		}
	}
	public String getTime(){		
		SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
		return df.format(new Date());
	}
 
	public long getTimeNet(String oldTime,String newTime){
		SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
		Date oldTime2 = null;
		Date newTime2 = null;
		try {
			oldTime2 = sdf.parse(oldTime);
			newTime2 = sdf.parse(newTime);
		} catch (ParseException e) {
			e.printStackTrace();
		}
		return Math.abs(newTime2.getTime() - oldTime2.getTime());
	}
 
	public static void main(String[] args) {
		QTPMonitor qtpm = new QTPMonitor();
		qtpm.start();
	}
}

Operations.java

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.text.SimpleDateFormat;
import java.util.Date;
 
public class Operations {
	public static boolean killTask(String processName){
		String pids = getTaskPIDs(processName);
		if(killTaskByName(processName)){
			String[] pid = pids.split(";");
			for(int i=0;i<pid .length;i++){
				if(findTaskByPID(pid[i])){
					System.out.println("kill task "+ processName +" failed, retry!");
					return false;
				}
			}
			return true;
		}else{
			return false;
		}
	}
	public static boolean findTaskByName(String processName) {  
		BufferedReader bufferedReader = null;  
		try {  
			Process proc = Runtime.getRuntime().exec("tasklist  /FI \"IMAGENAME eq " + processName + "\" /FO CSV /NH"); 
			bufferedReader = new BufferedReader(new InputStreamReader(proc.getInputStream()));  
			String line = null;  
			while ((line = bufferedReader.readLine()) != null) {  
				if (line.contains(processName)) {  
					return true;
				}         
			}  
			return false;  
		} catch (Exception ex) {  
			ex.printStackTrace();  
			return false;  
		} finally {  
			if (bufferedReader != null) {  
				try {  
					bufferedReader.close();  
				} catch (Exception ex) {}  
			}  
		}  
	} 
 
	public static boolean killTaskByName(String taskName){
		try {
			Runtime.getRuntime().exec("Taskkill /IM "+taskName+" /T /F");
			try {
				Thread.sleep(5000);
			} catch (InterruptedException e) {
				e.printStackTrace();
			}
		} catch (IOException e) {
			e.printStackTrace();
			return false;
		}
		return true;
	}
	public static int getTaskMemory(String processName) {  
		BufferedReader bufferedReader = null;  
		int rtn=0;
		try {  
			Process proc = Runtime.getRuntime().exec("tasklist  /FI \"IMAGENAME eq " + processName + "\" /FO CSV /NH"); 
			bufferedReader = new BufferedReader(new InputStreamReader(proc.getInputStream()));  
			String line = null;  
			while ((line = bufferedReader.readLine()) != null) {  
				if (line.contains(processName)) {
						String[] row =  line.split("\"");
						rtn = Integer.parseInt(row[row.length-1].split("K")[0].trim().replace(",", ""));
						System.out.println("the memory "+processName+" used is "+rtn+" KB");
				}          
			}  			
		} catch (Exception ex) {  
			ex.printStackTrace();   
		} finally {  
			if (bufferedReader != null) {  
				try {  
					bufferedReader.close();  
				} catch (Exception ex) {}  
			}  
		} 
		return rtn;  
	} 
	public static boolean findTaskByPID(String pid) {  
		BufferedReader bufferedReader = null;  
		try {  
			Process proc = Runtime.getRuntime().exec("tasklist  /FI \"PID eq " + pid + "\" /FO CSV /NH"); 
			bufferedReader = new BufferedReader(new InputStreamReader(proc.getInputStream()));  
			String line = null;  
			while ((line = bufferedReader.readLine()) != null) {  
				if (line.contains(pid)) {  
					return true;
				}          
			}  
			return false;  
		} catch (Exception ex) {  
			ex.printStackTrace();  
			return false;  
		} finally {  
			if (bufferedReader != null) {  
				try {  
					bufferedReader.close();  
				} catch (Exception ex) {}  
			}  
		}  
	} 
 
	public static String getTaskPIDs(String processName) {  
		BufferedReader bufferedReader = null;  
		String rtn="";
		try {  
			Process proc = Runtime.getRuntime().exec("tasklist  /FI \"IMAGENAME eq " + processName + "\" /FO CSV /NH"); 
			bufferedReader = new BufferedReader(new InputStreamReader(proc.getInputStream()));  
			String line = null;  
			while ((line = bufferedReader.readLine()) != null) {  
				if (line.contains(processName)) {
					if(rtn !=""){
						rtn = rtn +";"+line.split(",")[1].split("\"")[1];
					}else{
						rtn = line.split(",")[1].split("\"")[1];
					}
				}          
			}  			
		} catch (Exception ex) {  
			ex.printStackTrace();   
		} finally {  
			if (bufferedReader != null) {  
				try {  
					bufferedReader.close();  
				} catch (Exception ex) {}  
			}  
		} 
		return rtn;  
	} 
 
	public static boolean killTaskByPID(String pid){
		try {
			Runtime.getRuntime().exec("Taskkill /pid "+pid+" /T /F");
			try {

				return true;
			}
		} catch (IOException e) {
			e.printStackTrace();
			return false;
		}
	}

		BufferedReader bufferedReader = null;  
		double rtn=0;
		try { 
			String command = "typeperf \"\\Process("+processName.split(".exe")[0]+")\\% Processor Time\" -sc 1";
			Process proc = Runtime.getRuntime().exec(command); 
			bufferedReader = new BufferedReader(new InputStreamReader(proc.getInputStream()));  
			String line = null;  
			while ((line = bufferedReader.readLine()) != null) {  
				if (line.contains(getTime())) {
						String[] row =  line.split("\"");

						System.out.println("the CPU "+processName+" used is "+rtn);
				}          
			}  			
		} catch (Exception ex) {  
			ex.printStackTrace();   
		} finally {  
			if (bufferedReader != null) {  
				try {  
					bufferedReader.close();  
				} catch (Exception ex) {}  
			}  
		} 
		return rtn;  
	}
